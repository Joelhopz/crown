101  {
102  u32 old_size = array::size(q._queue);
103 
104  array::resize(q._queue, capacity);
105 
106  if (q._read + q._size > old_size)
107  {
108  memmove(array::begin(q._queue) + capacity - (old_size - q._read), array::begin(q._queue) + q._read, (old_size - q._read) * sizeof(T));
109  q._read += (capacity - old_size);
110  }

164  template <typename T>
165  inline void push(Queue<T>& q, const T *items, u32 n)
166  {
167  if (q.space() < n)
168  q.grow(q.size() + n);
169 
170  const u32 size = array::size(q._queue);
171  const u32 insert = (q._read + q._size) % size;
172 
173  u32 to_insert = n;
174  if (insert + to_insert > size)
175  to_insert = size - insert;
176 
177  memcpy(array::begin(q._queue) + insert, items, to_insert * sizeof(T));
178 
179  q._size += to_insert;
180  items += to_insert;
181  n -= to_insert;
182  memcpy(array::begin(q._queue), items, n * sizeof(T));
183 
184  q._size += n;
185  }
