132 inline Quaternion power(const Quaternion& q, f32 exp)
133 {
134  if (fabs(q.w) < 0.9999)
135  {
136  const f32 alpha = facos(q.w); // alpha = theta/2
137  const f32 new_alpha = alpha * exp;
138  const f32 mult = fsin(new_alpha) / fsin(alpha);
139 
140  Quaternion r;
141  r.w = fcos(new_alpha);
142  r.x = q.x * mult;
143  r.y = q.y * mult;
144  r.z = q.z * mult;
145  return r;
146  }

149 }

186 inline Quaternion lerp(const Quaternion& a, const Quaternion& b, f32 t)
187 {
188  const f32 t1 = 1.0f - t;
189 
190  Quaternion r;
191 
192  if (dot(a, b) < 0.0f)
193  {
194  r.x = t1*a.x + t*-b.x;
195  r.y = t1*a.y + t*-b.y;
196  r.z = t1*a.z + t*-b.z;
197  r.w = t1*a.w + t*-b.w;
198  }
199  else
200  {
201  r.x = t1*a.x + t*b.x;
202  r.y = t1*a.y + t*b.y;
203  r.z = t1*a.z + t*b.z;
204  r.w = t1*a.w + t*b.w;
205  }
206 
207  return normalize(r);
208 }
