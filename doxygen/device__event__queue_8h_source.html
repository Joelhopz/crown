170  bool push_event(const OsEvent& ev)
171  {
172  const int tail = _tail.load();
173  const int tail_next = (tail + 1) % MAX_OS_EVENTS;
174 
175  if (CE_UNLIKELY(tail_next == _head.load()))
176  return false;
177 
178  _queue[tail] = ev;
179  _tail.store(tail_next);
180  return true;
181  }
182 
183  bool pop_event(OsEvent& ev)
184  {
185  const int head = _head.load();
186 
187  if (CE_UNLIKELY(head == _tail.load()))
188  return false;
189 
190  ev = _queue[head];
191  _head.store((head + 1) % MAX_OS_EVENTS);
192 
193  return true;
194  }
