88  template <int BUFFER_SIZE>
89  void *TempAllocator<BUFFER_SIZE>::allocate(u32 size, u32 align)
90  {
91  _p = (char *)memory::align_top(_p, align);
92  if ((int)size > _end - _p) {
93  u32 to_allocate = sizeof(void *) + size + align;
94  if (to_allocate < _chunk_size)
95  to_allocate = _chunk_size;
96  _chunk_size *= 2;
97  void *p = _backing.allocate(to_allocate);
98  *(void **)_start = p;
99  _p = _start = (char *)p;
100  _end = _start + to_allocate;
101  *(void **)_start = 0;
102  _p += sizeof(void *);
103  _p = (char *)memory::align_top(_p, align);
104  }
105  void *result = _p;
106  _p += size;
107  return result;
108  }
