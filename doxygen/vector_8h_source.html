125  template <typename T>
126  inline void set_capacity(Vector<T>& v, u32 capacity)
127  {
128  if (capacity == v._capacity)
129  return;
130 
131  if (capacity < v._size)
132  resize(v, capacity);
133 
134  if (capacity > 0)
135  {
136  T* tmp = v._data;
137  v._capacity = capacity;
138  v._data = (T*)v._allocator->allocate(capacity * sizeof(T), alignof(T));
139 
140  for (u32 i = 0; i < v._size; ++i)
141  new (v._data + i) T(tmp[i]);
142 
143  for (u32 i = 0; i < v._size; ++i)
144  tmp[i].~T();
145 
146  v._allocator->deallocate(tmp);
147  }
148  }
