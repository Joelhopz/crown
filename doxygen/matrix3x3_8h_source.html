45 inline Matrix3x3 matrix3x3(const Quaternion& r)
46 {
47  Matrix3x3 m;
48  m.x.x = 1.0f - 2.0f * r.y * r.y - 2.0f * r.z * r.z;
49  m.x.y = 2.0f * r.x * r.y + 2.0f * r.w * r.z;
50  m.x.z = 2.0f * r.x * r.z - 2.0f * r.w * r.y;
51 
52  m.y.x = 2.0f * r.x * r.y - 2.0f * r.w * r.z;
53  m.y.y = 1.0f - 2.0f * r.x * r.x - 2.0f * r.z * r.z;
54  m.y.z = 2.0f * r.y * r.z + 2.0f * r.w * r.x;
55 
56  m.z.x = 2.0f * r.x * r.z + 2.0f * r.w * r.y;
57  m.z.y = 2.0f * r.y * r.z - 2.0f * r.w * r.x;
58  m.z.z = 1.0f - 2.0f * r.x * r.x - 2.0f * r.y * r.y;
59  return m;
60 }
